cmake_minimum_required(VERSION 3.6)
set(CMAKE_CONFIGURATION_TYPES Debug Release CACHE TYPE INTERNAL FORCE)
project(StereoReconstruction)
set(CMAKE_CXX_STANDARD 14)

# require some packages, include the directory of the package
find_package(OpenCV REQUIRED)
find_package(Eigen3 REQUIRED)
find_package(Flann REQUIRED)
find_package(Ceres REQUIRED)
include_directories("/opt/homebrew/Cellar/opencv/4.6.0_1/include/opencv4")
include_directories("/opt/homebrew/Cellar/eigen/3.4.0_1/include/eigen3")
include_directories("/opt/homebrew/Cellar/flann/1.9.1_13/include/flann")
include_directories("/opt/homebrew/Cellar/ceres-solver/2.1.0_1/include/ceres")

# link libraries
add_executable(StereoReconstruction main.cpp)
target_link_libraries(StereoReconstruction ${OpenCV_LIBS})
target_link_libraries(StereoReconstruction ${Eigen3_LIBS})
target_link_libraries(StereoReconstruction ${Flann_LIBS})
target_link_libraries(StereoReconstruction ${Ceres_LIBS})
